    const normalizeDateStr = (dateVal) => {

      if (!dateVal) return '';

      const dateStr = dateVal.toString().trim();

      if (/^\d{4}-\d{2}-\d{2}$/.test(dateStr)) return dateStr;

      

      try {

        const d = new Date(dateStr);

        if (!isNaN(d.getTime())) {

          const y = d.getFullYear();

          const m = String(d.getMonth() + 1).padStart(2, '0');

          const day = String(d.getDate()).padStart(2, '0');

          return `${y}-${m}-${day}`;

        }

      } catch(e) {}


      if (/^\d{4}-\d{2}-\d{2}/.test(dateStr)) {

        return dateStr.substring(0, 10);

      }

      return dateStr;

    };


    const normalizeTimeStr = (timeVal) => {

      if (!timeVal) return '';

      const timeStr = timeVal.toString().trim();

      

      if (/^\d{1,2}:\d{2}$/.test(timeStr)) {

        const parts = timeStr.split(':');

        return `${parts[0].padStart(2, '0')}:${parts[1]}`;

      }

      if (/^\d{1,2}:\d{2}:\d{2}/.test(timeStr)) {

        return timeStr.substring(0, 5);

      }


      try {

        const d = new Date(timeStr);

        if (!isNaN(d.getTime())) {

          const h = String(d.getHours()).padStart(2, '0');

          const m = String(d.getMinutes()).padStart(2, '0');

          return `${h}:${m}`;

        }

      } catch(e) {}


      const timeMatch = timeStr.match(/(\d{1,2}):(\d{2})/);

      if (timeMatch) {

        return `${timeMatch[1].padStart(2, '0')}:${timeMatch[2]}`;

      }


      return timeStr.substring(0, 5);

    };


    const formatTimeMalay = (timeStr) => {

      if (!timeStr) return '';

      const upper = timeStr.toString().trim().toUpperCase();

      

      if (upper.includes('PAGI') || upper.includes('PETANG') || upper.includes('MALAM') || upper.includes('TENGAH HARI')) {

        return upper;

      }


      const match = upper.match(/^(\d{1,2}):(\d{2})/);

      if (!match) return upper;


      let hours = parseInt(match[1], 10);

      const minutes = match[2];


      let period = 'PAGI';

      if (hours >= 12 && hours < 14) {

        period = 'TENGAH HARI';

      } else if (hours >= 14 && hours < 19) {

        period = 'PETANG';

      } else if (hours >= 19 && hours <= 23) {

        period = 'MALAM';

      } else if (hours === 0) {

        period = 'TENGAH MALAM';

      } else {

        period = 'PAGI';

      }


      let displayHours = hours % 12;

      if (displayHours === 0) {

        displayHours = 12;

      }

      const padHours = String(displayHours).padStart(2, '0');


      return `${padHours}:${minutes} ${period}`;

    };


    const parseDateTime = (dateStr, timeStr) => {

      if (!dateStr || !timeStr) return null;

      const [year, month, day] = dateStr.split('-').map(Number);

      const [hours, minutes] = timeStr.split(':').map(Number);

      return new Date(year, month - 1, day, hours, minutes, 0, 0);

    };
